template<size_t N, typename T>
struct utils::gen_tuple_t< N, T >

Генерация типа гомогенного кортежа, содержащего N элементов типа T.

Параметры шаблона
N- размер кортежа.
T- тип элементов.

Определения типов

template<size_t N, typename T>
using utils::gen_tuple_t< N, T >::type = decltype(std::tuple_cat(std::declval<std::tuple<T>>(), std::declval<typename gen_tuple_t<N - 1, T>::type>()))
